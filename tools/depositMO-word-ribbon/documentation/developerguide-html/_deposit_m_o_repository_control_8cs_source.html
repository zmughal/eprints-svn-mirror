00144         {

00146             // increment the counter here, and decrement it at every handled exit point of the method
00147             Globals.Word2010DepositMOAddIn.IncrementSubmissionThreadCount();
00148 
00149             #region Pre-submission checks
00150             // pre-submission checks
00151             if (String.IsNullOrWhiteSpace(endpoint))
00152             {
00153                 this.WarningBox("Attempting to submit document to empty endpoint; cowardly refusing to continue");
00154                 Globals.Word2010DepositMOAddIn.DecrementSubmissionThreadCount();
00155                 return false;
00156             }
00157             #endregion
00158 
00159             // grab a reference to the active document outside the lock so if someone
00160             // changes the active document whilst another thread is active it will
00161             // be the correct document
00162             Document d = Globals.Word2010DepositMOAddIn.Application.ActiveDocument;
00163             
00164             // ensure the user has saved their document before continuing (see
00165             // read-lock comments below)
00166             try
00167             {
00168                 d.Save();
00169             }
00170             catch
00171             {
00172                 // for some reason the standard flow control of Word means that
00173                 // if the user cancels the save dialogue, an exception will be thrown
00174                 // --- this is contrary to standard Windows Forms behaviour                
00175                 this.WarningBox("Didn't save the document this time; deposition aborted");
00176                 Globals.Word2010DepositMOAddIn.DecrementSubmissionThreadCount();
00177                 return false;
00178             }
00179             
00180             // this could be spun off on a separate (singleton-type) thread
00181             // to avoid UI glitches if the remote isn't responding quickly 
00182             lock (lockobj)
00183             {                
00184                 SwordClientHandler sch = null;                                
00185                 string originalDocumentName = null;
00186                 string tempFileName = null;
00187                 bool updateResult = false;
00188 
00189                 // get the full path and filename of the document and remember this
00190                 originalDocumentName = d.FullName;
00191 
00192                 // save the file to the temporary path
00193                 tempFileName += Path.GetTempPath() + d.Name;
00194 
00195                 try
00196                 {
00197                     // we need to 'save as' under a different filename as Word has an exclusive
00198                     // read-lock on the current document; 'Type.Missing' through the call until
00199                     // we can set the 'add to recent files' flag to false
00200                     d.SaveAs2(tempFileName, Type.Missing, Type.Missing, Type.Missing, false);
00201 
00202                     // we then need to 'save as' the original filename to restore the control
00203                     // to the user
00204                     d.SaveAs2(originalDocumentName);
00205                 }
00206                 catch (COMException cex)
00207                 {                    
00208                     this.logMessage("A problem occurred with " + cex.Source + " whilst trying to save the document");
00209                     this.logException(cex);
00210                     this.logMessage("Couldn't submit document " + originalDocumentName + " this time as " + cex.Source + " will not allow it");                    
00211                     Globals.Word2010DepositMOAddIn.DecrementSubmissionThreadCount();
00212                     return false;
00213                 }
00214 
00215                 // initialise the SWORD handler
00216                 this.logMessage("Connecting to repository at " + endpoint);
00217                 sch = new SwordClientHandler(username, password, endpoint);
00218 
00219                 // use the PUT parts of SWORD if we're updating
00220                 if (updateExisting)
00221                 {
00222                     // 'endpoint' in this context should now be the location of the document
00223                     #region Update existing document
00224                     try { updateResult = sch.PutToDocument(tempFileName, endpoint); }
00225                     catch (Exception ex)
00226                     {
00227                         this.logMessage("Could not update document at " + endpoint);
00228                         this.logException(ex);
00229                     }
00230                     finally { File.Delete(tempFileName); }
00231 
00232                     if (updateResult)
00233                     {
00234                         this.logMessage("Successfully updated document at " + endpoint);
00235                         Globals.Word2010DepositMOAddIn.DecrementSubmissionThreadCount();
00236                         return true;
00237                     }
00238                     else
00239                     {
00240                         this.logMessage("Couldn't update document at " + endpoint);
00241                         Globals.Word2010DepositMOAddIn.DecrementSubmissionThreadCount();
00242                         return false;
00243                     }
00244                     #endregion
00245                 }
00246                 else
00247                 {
00248                     #region Create new document
00249                     // create new document
00250                     string documentUri = null;
00251                     
00252                     try { documentUri = sch.PostDocxToRepository(tempFileName); }
00253                     catch (Exception ex)
00254                     {
00255                         this.logMessage("Could not deposit document to " + endpoint);
00256                         this.logException(ex);
00257                     }
00258                     finally { File.Delete(tempFileName); }
00259                     #endregion
00260 
00261                     // check for null response. Behaviour at this point is undefined
00262                     if (documentUri == null)
00263                     {
00264                         this.logMessage("Couldn't understand response from server at endpoint " + endpoint + "; please check deposition manually");
00265                         Globals.Word2010DepositMOAddIn.DecrementSubmissionThreadCount();
00266                         return false;
00267                     }
00268 
00269                     #region Response handling on new deposition
00270                     // attempt to parse the response
00271                     if (!String.IsNullOrEmpty(documentUri))
00272                     {
00273                         this.logMessage("Document deposited at " + documentUri);
00274                         Globals.Word2010DepositMOAddIn.SetSingleLineTextBox(documentUri, this.documentLocationTextBox);
00275                     }
00276                     else
00277                     {
00278                         this.logMessage("Warning: response from endpoint " + endpoint + " did not contain a reference to the submitted document");
00279                         Globals.Word2010DepositMOAddIn.DecrementSubmissionThreadCount();
00280                         return false;
00281                     }
00282                     Globals.Word2010DepositMOAddIn.DecrementSubmissionThreadCount();
