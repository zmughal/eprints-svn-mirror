00306         private void AddRepositoryControl()
00307         {         
00308             if (this.Application.Documents.Count == 0)
00309             {
00310                 // COM will throw an exception if we try to do anything with ActiveWindow (even a test for null)
00311                 // if there are no open documents
00312                 return;
00313             }
00314 
00315             foreach (CustomTaskPane ctp in this.CustomTaskPanes)
00316             {
00317                 if (ctp != null && ctp.Window != null && ctp.Control != null && !ctp.Control.Disposing && ctp.Window == this.Application.ActiveWindow)
00318                 {
00319                     // we already have a control for this task pane so don't continue to add a new one
00320                     return;
00321                 }
00322             }        
00323             
00324             // create a new repository control, add it to the internal list and add it to
00325             // a custom task pane associated with this document window
00326             DepositMORepositoryControl dmrc = new DepositMORepositoryControl();
00327             this.repositoryControls.Add(dmrc);
00328             this.CustomTaskPanes.Add(dmrc, "Repository control", this.Application.ActiveWindow);            
00329         }
