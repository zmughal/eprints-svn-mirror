00307         private void AddRepositoryControl()
00308         {         
00309             if (this.Application.Documents.Count == 0)
00310             {
00311                 // COM will throw an exception if we try to do anything with ActiveWindow (even a test for null)
00312                 // if there are no open documents
00313                 return;
00314             }
00315 
00316             foreach (CustomTaskPane ctp in this.CustomTaskPanes)
00317             {
00318                 try
00319                 {
00320                     if (ctp != null && ctp.Window != null && ctp.Control != null && !ctp.Control.Disposing && ctp.Window == this.Application.ActiveWindow)
00321                     {
00322                         // we already have a control for this task pane so don't continue to add a new one
00323                         return;
00324                     }
00325                 }
00326                 catch (COMException cex)
00327                 {
00328                     this.LogMessage("Could not check for taskpane addition; full COM exception follows:");
00329                     this.LogMessage(cex.ToString());
00330                     this.LogMessage("You may wish to restart Microsoft Word");
00331                     return;
00332                 }
00333             }        
00334             
00335             // create a new repository control, add it to the internal list and add it to
00336             // a custom task pane associated with this document window
00337             DepositMORepositoryControl dmrc = new DepositMORepositoryControl();
00338             this.repositoryControls.Add(dmrc);
00339             this.CustomTaskPanes.Add(dmrc, "Repository control", this.Application.ActiveWindow);            
00340         }
